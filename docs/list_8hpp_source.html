32 struct foldr_op {
33 template <class Fn, std::movable T, std::input_iterator I, std::sentinel_for<I> S>
34 constexpr auto operator()(Fn op, T&& init, I first, S last) const -> T {
35 if (first == last) {
36 return std::forward<T>(init);
37 } else {
38 return std::invoke(
39 op, *first,
40 this->operator()(op, std::forward<T>(init), std::ranges::next(first), last));
41 }
42 }
