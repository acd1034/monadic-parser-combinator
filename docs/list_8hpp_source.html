43 struct foldr_op {
44 template <class Fn, std::movable T, std::input_iterator I, std::sentinel_for<I> S>
45 constexpr auto operator()(Fn op, T&& init, I first, S last) const -> T {
46 if (first == last) {
47 return std::forward<T>(init);
48 } else {
49 auto tmp = first++;
50 return std::invoke(op, *tmp, this->operator()(op, std::forward<T>(init), first, last));
51 }
52 }
