11 // copyable_box allows turning a type that is copy-constructible (but maybe not copy-assignable)
12 // into a type that is both copy-constructible and copy-assignable. It does that by introducing an
13 // empty state and basically doing destroy-then-copy-construct in the assignment operator. The
14 // empty state is necessary to handle the case where the copy construction fails after destroying
15 // the object.
16 //
17 // In some cases, we can completely avoid the use of an empty state; we provide a specialization
18 // of copyable_box that does this, see below for the details.
19
21 template <class T>
22 concept copy_constructible_object = std::copy_constructible<T> and std::is_object_v<T>;

87 // This partial specialization implements an optimization for when we know we don't need to store
88 // an empty state to represent failure to perform an assignment. For copy-assignment, this
89 // happens:
90 //
91 // 1. If the type is copyable (which includes copy-assignment), we can use the type's own
92 // assignment operator
93 // directly and avoid using std::optional.
94 // 2. If the type is not copyable, but it is nothrow-copy-constructible, then we can implement
95 // assignment as
96 // destroy-and-then-construct and we know it will never fail, so we don't need an empty state.
97 //
98 // The exact same reasoning can be applied for move-assignment, with copyable replaced by movable
99 // and nothrow-copy-constructible replaced by nothrow-move-constructible. This specialization is
100 // enabled whenever we can apply any of these optimizations for both the copy assignment and the
101 // move assignment operator.
102
104 template <class T>
105 concept doesnt_need_empty_state_for_copy =
106 std::copyable<T> or std::is_nothrow_copy_constructible_v<T>;
107
108 template <class T>
109 concept doesnt_need_empty_state_for_move =
110 std::movable<T> or std::is_nothrow_move_constructible_v<T>;
112
114 template <copy_constructible_object T>
115 requires doesnt_need_empty_state_for_copy<T> and doesnt_need_empty_state_for_move<T>

153 std::construct_at(std::addressof(instance_), std::move(other.instance_));
